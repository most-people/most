## 三、`MostToken.sol` 核心设计

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract MostToken is ERC20, Ownable {
    uint256 public constant TOTAL_SUPPLY = 21_000_000 * 10**18;

    // 分配比例
    uint256 public constant MINING_POOL = 40;      // 40% 节点质押激励
    uint256 public constant TEAM_ALLOCATION = 10;  // 10% 团队
    uint256 public constant COMMUNITY_ALLOCATION = 30; // 30% 社区生态
    uint256 public constant RESERVE_ALLOCATION = 20;   // 20% 储备

    // 销毁统计
    uint256 public totalBurned;

    constructor() ERC20("MostBox Token", "MOST") {
        _mint(msg.sender, TOTAL_SUPPLY);
    }

    // 销毁函数（手续费购买销毁、罚没销毁）
    function burn(uint256 amount) external {
        _burn(msg.sender, amount);
        totalBurned += amount;
    }

    // 分配释放（由 keeper 调用）
    function releaseAllocation(address recipient, uint256 amount) external onlyOwner {
        _transfer(msg.sender, recipient, amount);
    }
}
```

## 四、StorageMarket 合约（稳定币支付）

### 4.1 核心数据结构

```solidity
contract StorageMarket {
    // 支持的稳定币
    IERC20 public constant USDC = IERC20(0x...);
    IERC20 public constant USDT = IERC20(0x...);

    struct StorageOrder {
        bytes32 orderId;
        address user;
        bytes32 contentCid;      // 加密后内容的 CID
        uint256 fileSize;        // 文件大小（bytes）
        uint256 duration;        // 存储时长（秒）
        uint256 redundancy;      // 冗余副本数
        uint256 price;           // 总价格（稳定币，6 位小数）
        uint256 createdAt;
        OrderStatus status;
        address[] storageNodes;
    }

    struct NodeInfo {
        address nodeAddress;
        uint256 mostStake;       // MOST 质押量
        uint256 totalStorage;    // 总存储量
        uint256 onlineTime;      // 在线时长
        uint256 reputation;      // 信誉分
        uint256 lastProofTime;   // 最后证明时间
        bool isActive;
    }

    // 核心函数
    function createOrder(bytes32 cid, uint256 size, uint256 duration, uint256 redundancy, address stablecoin) external;
    function registerNode() external;  // 需先质押 MOST
    function submitProof(bytes32 orderId, bytes memory proof) external;
    function claimRewards() external;  // 领取稳定币收益
    function slashNode(address node, uint256 penalty) external;
}
```

### 4.2 订单匹配流程

```
1. 用户调用 createOrder() + 支付 USDC/USDT
2. 合约根据节点资质（MOST 质押量、信誉、存储余量）匹配
3. 合约触发 OrderMatched 事件
4. 节点监听事件，准备接收数据
5. 用户通过 P2P 发送加密分片
6. 节点接收后提交存储证明
7. 合约验证证明，订单生效
8. 用户支付的稳定币 90% 归节点，10% 进入 FeeDistributor
```

### 4.3 定价与收益

| 机制           | 规则                                                     |
| -------------- | -------------------------------------------------------- |
| **稳定币定价** | $0.01/GB/月（动态调整）                                  |
| **收益分配**   | 节点 90% / 网络手续费 10%                                |
| **节点等级**   | 按 MOST 质押量、在线时长、信誉分级，高等级优先匹配       |
| **惩罚机制**   | 离线 >24h 扣当日收益；丢失数据扣 MOST 质押；多次违规封禁 |

### 4.4 定价公式

```
基础价格 = 存储量(GB) × 时长(天) × 单价(GB/天)

单价 = 基准单价($0.01) × 供需系数 × 节点等级系数

供需系数 = 全网总需求存储量 / 全网总可用存储量
  - > 1.0: 资源紧张，价格上涨
  - < 1.0: 资源充足，价格下降

节点等级系数:
  - Lv1 (新节点): 1.0
  - Lv2 (信誉 > 80): 0.95 (5% 折扣吸引订单)
  - Lv3 (信誉 > 95): 0.90 (10% 折扣)
  - Lv4 (顶级节点): 0.85 (15% 折扣)
```

## 五、FeeDistributor 合约（手续费分配+MOST 销毁）

```solidity
contract FeeDistributor {
    IERC20 public mostToken;
    IERC20 public usdc;
    IUniswapV2Router public uniswapRouter;

    // 分配手续费
    function distributeFees(address stablecoin, uint256 amount) external {
        // 50% 稳定币归网络储备
        // 50% 稳定币通过 DEX 购买 MOST 并销毁
        uint256 halfAmount = amount / 2;

        // 储备部分
        stablecoin.transfer(reserveAddress, halfAmount);

        // 购买并销毁 MOST
        uniswapRouter.swapExactTokensForTokens(
            halfAmount,
            0,
            [stablecoin, mostToken],
            address(this),
            block.timestamp
        );
        mostToken.burn(mostToken.balanceOf(address(this)));
    }

    // 罚没销毁
    function burnSlashed(uint256 mostAmount) external {
        mostToken.burn(mostAmount);
    }
}
```

## 六、正向循环

```
用户支付 USDC → 节点获得 90% 稳定币收入（无波动风险）
                    ↓
              10% 手续费 → FeeDistributor
                    ↓
              50% 归网络储备 + 50% 购买并销毁 MOST
                    ↓
              MOST 总量减少 → MOST 增值
                    ↓
              节点质押 MOST 更有价值 → 更多节点加入
                    ↓
              网络容量增加 → 更多用户 → 更多销毁
```
